# Encoder-LSTM-Decoder with causal convolutions + skip connections

class CRN(nn.Module):
    def forward(self, x: torch.Tensor):
        # x: (B, 2, F=257, T)  — stacked real/imag
        feat = x
        skip_connections = []

        for enc_layer in self.encoder:          # 5 causal conv blocks
            feat = enc_layer(feat)
            skip_connections.append(feat)

        # LSTM on (B, T, C×F) sequence
        feat_seq  = feat.permute(0,3,1,2).reshape(B, T, -1)
        lstm_out, _ = self.lstm(feat_seq)
        feat_seq  = self.lstm_proj(lstm_out)
        feat      = feat_seq.reshape(B, T, C, F).permute(0,2,3,1)

        for i, dec_layer in enumerate(self.decoder):   # U-Net skip concat
            feat = torch.cat([feat, skip_connections[-(i+1)]], dim=1)
            feat = dec_layer(feat)

        mask = torch.tanh(feat)   # CRM bounded in (-1, 1)
        return mask[:, 0], mask[:, 1]   # mask_real, mask_imag

    def enhance(self, noisy_real, noisy_imag):
        mr, mi = self.forward(torch.stack([noisy_real, noisy_imag], 1))
        # Complex multiplication: (mr + j·mi) ⊗ (nr + j·ni)
        enh_real = mr * noisy_real - mi * noisy_imag
        enh_imag = mr * noisy_imag + mi * noisy_real
        return enh_real, enh_imag
